Example #1

Unordered list container with list items containing img tags.

HTML

<ul class='gallery-container-1'>
  <li class='slides'>
    <img class='image-slide' src='images/1.jpg'/>
  </li>
  <li class='slides'>
    <img class='image-slide' src='images/2.jpg'/>
  </li>
  <li class='slides'>
    <img class='image-slide' src='images/3.jpg'/>
  </li>
  <li class='slides'>
    <img class='image-slide' src='images/4.jpg'/>
  </li>
  <li class='slides'>
    <img class='image-slide' src='images/5.jpg'/>
  </li>
</ul>

JS

$('ul.gallery-container-1').owlgallery({
  child: 'li.slides',
  galleryWidth: 1000,
  galleryHeight: 563,
  enableTweener: true
});

Example #2

Div container with a list of divs containing an img tags. Set to fading instead of sliding.

HTML

<div class='gallery-container-2'>
	<div class='slides'>
		<img class='image-slide' src='images/1.jpg'/>
	</div>
	<div class='slides'>
		<img class='image-slide' src='images/2.jpg'/>
	</div>
	<div class='slides'>
		<img class='image-slide' src='images/3.jpg'/>
	</div>
	<div class='slides'>
		<img class='image-slide' src='images/4.jpg'/>
	</div>
	<div class='slides'>
		<img class='image-slide' src='images/5.jpg'/>
	</div>
</div>

JS

$('div.gallery-container-2').owlgallery({
	child: 'div.slides',
	cycleTime: 3000,
	animationTime: 1000,
	galleryWidth: 1000,
	galleryHeight: 563,
	animationTypes: Owl.animationTypes.FADE,
	enableTweener: true
});

Example #3

Div container with a list of img tags. Pagination and nav controls.

HTML

<div class='gallery-container-3'>
	<div class='image-container' >
		<img class='image-slide' src='images/1.jpg'/>
		<img class='image-slide' src='images/2.jpg/'>
		<img class='image-slide' src='images/3.jpg'/>
		<img class='image-slide' src='images/4.jpg'/>
		<img class='image-slide' src='images/5.jpg'/>
	</div>
	<div class='nav'>
		<div class='sprite arrow-left'></div>
		<div class='sprite arrow-right'></div>
	</div>
	<div class='pagination'>
		<div class='sprite dot-empty'></div>
	</div>
</div>

JS

$('div.gallery-container-3').owlgallery({
	child: 'img.image-slide',
	cycleTime: 4500,
	animationTime: 1000,
	galleryWidth: 1000,
	galleryHeight: 563,
	enableTweener: true,
	responsiveMode: Owl.responsiveMode.ONLYRESIZEWHENSMALLER,
	paginationElement: '.pagination',
	navElement: '.nav'
});
