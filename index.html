Example #2

Div container with a list of divs containing an img tags. Set to fading instead of sliding.

HTML

<div class='gallery-container-2'>
  <div class='slides'>
  <img class='image-slide' src='images/1.jpg'/>
  </div>
  <div class='slides'>
  <img class='image-slide' src='images/2.jpg'/>
  </div>
  <div class='slides'>
  <img class='image-slide' src='images/3.jpg'/>
  </div>
  <div class='slides'>
    <img class='image-slide' src='images/4.jpg'/>
  </div>
  <div class='slides'>
    <img class='image-slide' src='images/5.jpg'/>
  </div>
</div>

JS

$('div.gallery-container-2').owlgallery({
  child: 'div.slides',
  cycleTime: 3000,
  animationTime: 1000,
  galleryWidth: 1000,
  galleryHeight: 563,
  animationTypes: Owl.animationTypes.FADE,
  enableTweener: true
});
